begin;

  delete from public.quotes;   -- saved quotes
  delete from public.agents;   -- realtor agents
  delete from public.events;   -- activity log (quotes run, notes, captures)

-- verify: test data cleared, logins + tenants kept

union all select 'tenants (kept)',   count(*) from public.tenants
